      obj.ShowToolBar = false ; //隐藏工具条
     
      obj.JSEnv = 1;
      obj.ShowScrollBarButton = 1;
      obj.LoadFileBase64(content);
    }
    </ SCRIPT >

aip转化word到aip
LoadFile

  打开文件,支持所有的文件格式。
　　AIP文件自动打开。
　　非AIP文件自动转化,主动生成书签和把原始文件包含在生成的AIP文件中,如想控制书签和原始文件的装载,请参阅LoadFileEx。

long LoadFile(
　　　　LPCTSTR pcFileName, 　　　//文件路径。
　　　　);

参数

pcFileName 要打开的文件路径。

返回值

0 失败。
1 正确。
示例
Object.LoadFile("") 将打开一个选择文件的对话框
Object.LoadFile("C:\aip.aip") 打开C:\aip.aip文件
Object.LoadFile("C:\doc.doc") 打开C:\doc.doc文件
