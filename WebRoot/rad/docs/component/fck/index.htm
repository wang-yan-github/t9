适用情景描述 特别强调 代码
简单调用 引用js:<%=contextPath %>/core/js/cmp/fck/fckeditor/fckeditor.js
创建：
// fck所在目录.
var sBasePath = contextPath + "/core/js/cmp/fck/fckeditor/" ;
//实例化一个fck
var oFCKeditor = new FCKeditor( 'FCKeditor1' ) ;
oFCKeditor.BasePath = sBasePath ;
//设置fck高度
oFCKeditor.Height = 300 ;
//设置fck默认值
oFCKeditor.Value  = '<p>This is some <strong>sample text<\/strong>. You are using <a href="http://www.fckeditor.net/">FCKeditor<\/a>.<\/p>' ;
oFCKeditor.Create() ;
 具体代码请参见：
rad/docs/component/fck/_samples/html/sample01.jsp
配置工具栏 配置fckconfig.js:
以增加自定义"feedback"工具栏为例。在fckconfig.js中加入:FCKConfig.ToolbarSets["feedback"] = [['FontName','FontSize','Bold','Italic','TextColor','BGColor','-','OrderedList','UnorderedList','-','Link','Unlink','-','Smiley','clearEdit']] ;
设置工具栏：
oFCKeditor.ToolbarSet = "feedback";
取得fck工具栏名称：
oFCKeditor.ToolbarSet.Name 		具体代码请参见：
rad/docs/component/fck/_samples/html/sample04.jsp
其它相关配置 配置皮肤:
//sSkin 为相应的皮肤名,目前只提供两种皮肤"office2003","silver"
var sSkinPath = sBasePath + 'editor/skins/' + sSkin + '/' ;
oFCKeditor.Config[ 'SkinPath' ] = sSkinPath ;
 
// The following switch is optional. It is done to enhance the loading
// time of the toolbar, by preloading the images used on it.
switch ( sSkin )
{
case 'office2003' :
oFCKeditor.Config[ 'PreloadImages' ] =
sSkinPath + 'images/toolbar.start.gif' + ';' +
sSkinPath + 'images/toolbar.end.gif' + ';' +
sSkinPath + 'images/toolbar.bg.gif' + ';' +
sSkinPath + 'images/toolbar.buttonarrow.gif' ;
break ;
 
case 'silver' :
oFCKeditor.Config[ 'PreloadImages' ] =
sSkinPath + 'images/toolbar.start.gif' + ';' +
sSkinPath + 'images/toolbar.end.gif' + ';' +
sSkinPath + 'images/toolbar.buttonbg.gif' + ';' +
sSkinPath + 'images/toolbar.buttonarrow.gif' ;
break ;
}
设置内容为整个页面：
oFCKeditor.Config['FullPage'] = true ;
设置fck语言：
oFCKeditor.ToolbarSet.Name
设置fck插件：
oFCKeditor.Config['CustomConfigurationsPath'] = sBasePath + '_samples/html/assets/sample06.config.js' ;
设置工具栏是否展开:
oFCKeditor.Config['ToolbarStartExpanded'] = false ; 		具体 配置皮肤代码请参见：
rad/docs/component/fck/_samples/html/sample05.html
相关接口调用 取得fck:
var oEditor = FCKeditorAPI.GetInstance('FCKeditor1') ;
插入html:
if ( oEditor.EditMode == FCK_EDITMODE_WYSIWYG )
{
  oEditor.InsertHtml( '- This is som HTML -') ;
}

oEditor.GetXHTML( true )
设置内容：
