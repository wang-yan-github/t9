function doInit(){
  var selected = [{value:'1',text:'公告通知'}
    ,{value:'2',text:'内部邮件'}
    ,{value:'3',text:'日程安排'}];
  var disselected = [{value:'6',text:'工资上报'}
    ,{value:'7',text:'工作计划'}];

  new ExchangeSelectbox({containerId:'div'//容器Id
    , selectedArray:selected //选中的项
    , disSelectedArray:disselected //未选中的项
    ,title:'选择'//标题
    ,selectName:'nextProcess'
      ,selectedChange:exchangeHandler}); 
    
}
function exchangeHandler(ids){
  //ids为选中项的值以逗号连接的字符串
  $('selectValue').value = ids;
}

两级-备选选中 
样式表：[cssPath]/cmp/ExchangeSelect.css
Js文件：[contextPath]/core/js/cmp/ExchangeSelect1.0.js
function doInit(){
  var selected = [{value:'1',text:'公告通知',parentId:2}
  ,{value:'2',text:'内部邮件'},{value:'3',text:'日程安排'}];
  var disselected = [{id:1,text:'系统模块',isParent:true}
  ,{value:'6',text:'工资上报'}
  ,{value:'5',text:'车辆申请',parentId:1}
  ,{value:'7',text:'工作计划',parentId:1}
  ,{id:2,text:'功能模块',isParent:true}];

  new ExchangeSelectbox({containerId:'div'
    , selectedArray:selected
    , disSelectedArray:disselected 
    ,isOneLevel:false
    ,title:'选择'
    ,selectName:'nextProcess'
      ,selectedChange:exchangeHandler}); 
}
function exchangeHandler(ids){
  $('selectValue').value = ids;
}

两级时首先需加入参数isOneLevel并设置为false
另外对选项数组中每项的也有要求：

对于子选项则必须指定它的父级id，如果是一般选项不需设置父级id
对于父选项则必须指定它的Id以及isParent属性必须是true

参数说明 
样式表：[cssPath]/cmp/ExchangeSelect.css
Js文件：Js文件：[contextPath]/core/js/cmp/ExchangeSelect1.0.js
js对象:ExchangeSelectbox
构造函数参数:{containerId,selectedArray,disSelectedArray,isSort
  ,isOneLevel,title,selectName,selectedChange,titleText}
  containerId 字符串 互选框所依附的div的id值,不指定默认为body
  selectedArray 对象数组  初始化时默认选中的对象

   
  disSelectedArray 对象数组  初始化时默认没有选中的对象

  isSort boolean 是否排序
  isOneLevel boolean 指定控件把展示数据的层数，现控件只支持两级

  title 字符串 指定控件的标题,不指定默认不显示
  selectName 字符串 指定控件选中框的name属性，主要方便form 表单中些控件的提交。

  titleText 标题的字{selectedTitle:'',disSelectedTitle:''}默认为选中,备选

  selectedChange 函数 指定选中的项改变时触发的事件处理函数
        
对el参数的说明：
如：本例中el是对应div的id值

对selectedArray和disSelectedArray对象数组的说明

  [{value:'',text:'',parentId:2,isParent:true,id:2},....]
  value:项的值

  text:项的label
  parentId:父项的id;
  id:如果是父项的话需指定id
  isParent:是否是父项
